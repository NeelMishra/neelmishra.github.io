    Design: Parking Lot

    March 2026 · 25 min read

    The parking lot is the hello world of LLD interviews. It tests your ability to identify entities, apply SOLID, and handle real-world constraints (different vehicle sizes, pricing strategies, concurrency). Let's build it from scratch.

    1Clarify Requirements

    Before writing any code, ask questions. Here's what we'll assume after clarification:

    Functional Requirements

    • Multiple floors, each with numbered spots
    • Three spot sizes: Small, Medium, Large
    • Three vehicle types: Motorcycle, Car, Truck
    • A motorcycle can park in any spot; a car needs medium or large; a truck needs large
    • Issue a ticket on entry with timestamp
    • Calculate fee on exit based on duration and vehicle type
    • Display available spots per floor

    Non-Functional

    • Thread-safe entry/exit (multiple gates)
    • Easy to add new vehicle types or pricing strategies

    4Define Enums and Data Types

    enum class VehicleType { MOTORCYCLE, CAR, TRUCK };

enum class SpotSize { SMALL, MEDIUM, LARGE };

// Which vehicles fit in which spots
bool canFit(VehicleType vt, SpotSize ss) {
    switch (vt) {
        case VehicleType::MOTORCYCLE: return true;  // fits anywhere
        case VehicleType::CAR:
            return ss == SpotSize::MEDIUM || ss == SpotSize::LARGE;
        case VehicleType::TRUCK:
            return ss == SpotSize::LARGE;
    }
    return false;
}

    5Implementation

    Vehicle Hierarchy OCP

    class Vehicle {
    string licensePlate;
    VehicleType type;
public:
    Vehicle(string plate, VehicleType t)
        : licensePlate(move(plate)), type(t) {}
    VehicleType getType() const { return type; }
    const string& getPlate() const { return licensePlate; }
    virtual ~Vehicle() = default;
};

class Car : public Vehicle {
public:
    explicit Car(string plate) : Vehicle(move(plate), VehicleType::CAR) {}
};

class Motorcycle : public Vehicle {
public:
    explicit Motorcycle(string plate) : Vehicle(move(plate), VehicleType::MOTORCYCLE) {}
};

class Truck : public Vehicle {
public:
    explicit Truck(string plate) : Vehicle(move(plate), VehicleType::TRUCK) {}
};

    Parking Spot SRP

    class ParkingSpot {
    int spotNumber;
    SpotSize size;
    Vehicle* parkedVehicle = nullptr;
public:
    ParkingSpot(int num, SpotSize sz) : spotNumber(num), size(sz) {}

    bool isAvailable() const { return parkedVehicle == nullptr; }
    SpotSize getSize() const { return size; }
    int getNumber() const { return spotNumber; }

    bool canFitVehicle(VehicleType vt) const {
        return isAvailable() && canFit(vt, size);
    }

    void park(Vehicle* v) { parkedVehicle = v; }
    Vehicle* unpark() {
        auto* v = parkedVehicle;
        parkedVehicle = nullptr;
        return v;
    }
};

    Parking Floor

    class ParkingFloor {
    int floorNumber;
    vector<ParkingSpot> spots;
public:
    ParkingFloor(int num, int small, int medium, int large)
        : floorNumber(num) {
        int id = 1;
        for (int i = 0; i < small; i++) spots.emplace_back(id++, SpotSize::SMALL);
        for (int i = 0; i < medium; i++) spots.emplace_back(id++, SpotSize::MEDIUM);
        for (int i = 0; i < large; i++) spots.emplace_back(id++, SpotSize::LARGE);
    }

    ParkingSpot* findSpot(VehicleType vt) {
        for (auto& spot : spots) {
            if (spot.canFitVehicle(vt)) return &spot;
        }
        return nullptr;
    }

    int available(SpotSize sz) const {
        return count_if(spots.begin(), spots.end(),
            [sz](const ParkingSpot& s) { return s.isAvailable() && s.getSize() == sz; });
    }

    int getFloorNumber() const { return floorNumber; }
};

    Ticket

    class Ticket {
    static int nextId;
    int id;
    Vehicle* vehicle;
    ParkingSpot* spot;
    int floorNumber;
    chrono::steady_clock::time_point entryTime;
public:
    Ticket(Vehicle* v, ParkingSpot* s, int floor)
        : id(nextId++), vehicle(v), spot(s), floorNumber(floor),
          entryTime(chrono::steady_clock::now()) {}

    Vehicle* getVehicle() const { return vehicle; }
    ParkingSpot* getSpot() const { return spot; }
    int getFloor() const { return floorNumber; }
    int getId() const { return id; }

    double hoursParked() const {
        auto now = chrono::steady_clock::now();
        auto mins = chrono::duration_cast<chrono::minutes>(now - entryTime).count();
        return max(1.0, ceil(mins / 60.0));  // minimum 1 hour
    }
};
int Ticket::nextId = 1;

    Pricing Strategy OCP DIP

    class PricingStrategy {
public:
    virtual ~PricingStrategy() = default;
    virtual double calculate(const Ticket& ticket) const = 0;
};

class HourlyPricing : public PricingStrategy {
    unordered_map<VehicleType, double> rates;
public:
    HourlyPricing() {
        rates[VehicleType::MOTORCYCLE] = 1.0;
        rates[VehicleType::CAR] = 2.0;
        rates[VehicleType::TRUCK] = 3.0;
    }
    double calculate(const Ticket& ticket) const override {
        auto it = rates.find(ticket.getVehicle()->getType());
        double rate = (it != rates.end()) ? it->second : 2.0;
        return ticket.hoursParked() * rate;
    }
};

class FlatPricing : public PricingStrategy {
    double flatRate;
public:
    explicit FlatPricing(double rate = 5.0) : flatRate(rate) {}
    double calculate(const Ticket& ticket) const override {
        return flatRate;
    }
};

    ParkingLot (Singleton + Facade) SRP

    class ParkingLot {
    vector<ParkingFloor> floors;
    unique_ptr<PricingStrategy> pricing;
    mutex mtx;  // thread-safe gate access

    ParkingLot() : pricing(make_unique<HourlyPricing>()) {}
public:
    static ParkingLot& getInstance() {
        static ParkingLot instance;
        return instance;
    }

    void addFloor(int small, int medium, int large) {
        floors.emplace_back(floors.size() + 1, small, medium, large);
    }

    void setPricing(unique_ptr<PricingStrategy> strategy) {
        pricing = move(strategy);
    }

    // Entry gate
    unique_ptr<Ticket> park(Vehicle* vehicle) {
        lock_guard<mutex> lock(mtx);
        for (auto& floor : floors) {
            ParkingSpot* spot = floor.findSpot(vehicle->getType());
            if (spot) {
                spot->park(vehicle);
                return make_unique<Ticket>(vehicle, spot, floor.getFloorNumber());
            }
        }
        return nullptr;  // lot is full for this vehicle type
    }

    // Exit gate
    double unpark(Ticket* ticket) {
        lock_guard<mutex> lock(mtx);
        double fee = pricing->calculate(*ticket);
        ticket->getSpot()->unpark();
        return fee;
    }

    void displayAvailability() const {
        for (const auto& floor : floors) {
            cout << "Floor " << floor.getFloorNumber() << ": "
                 << floor.available(SpotSize::SMALL) << " small, "
                 << floor.available(SpotSize::MEDIUM) << " medium, "
                 << floor.available(SpotSize::LARGE) << " large" << endl;
        }
    }
};

    6Putting It Together

    int main() {
    auto& lot = ParkingLot::getInstance();
    lot.addFloor(5, 10, 3);   // floor 1
    lot.addFloor(5, 10, 3);   // floor 2

    Car myCar("ABC-123");
    Motorcycle myBike("XYZ-789");

    auto ticket1 = lot.park(&myCar);
    auto ticket2 = lot.park(&myBike);

    lot.displayAvailability();

    if (ticket1) {
        double fee = lot.unpark(ticket1.get());
        cout << "Car fee: $" << fee << endl;
    }

    // Switch to flat pricing at night
    lot.setPricing(make_unique<FlatPricing>(3.0));
    if (ticket2) {
        double fee = lot.unpark(ticket2.get());
        cout << "Bike fee: $" << fee << endl;
    }
}

    7SOLID Review

    PrincipleHow we applied it
    SRPParkingSpot manages one spot. Ticket records a session. PricingStrategy calculates fees. No class does two jobs.
    OCPAdding a new vehicle type or pricing strategy = add a new class. No existing code changes.
    LSPCar, Motorcycle, Truck are all substitutable for Vehicle. HourlyPricing and FlatPricing are substitutable for PricingStrategy.
    ISPPricingStrategy has only one method. Vehicle has only getType() and getPlate(). No fat interfaces.
    DIPParkingLot depends on PricingStrategy (abstraction), not HourlyPricing (concrete). Strategy injected via setPricing().

    Common Extensions

    • Multiple entry/exit gates: Each gate calls park()/unpark() — the mutex handles concurrency.
    • Electric vehicle spots: Add SpotSize::EV and VehicleType::ELECTRIC.
    • Reservation system: Add a ReservationManager that pre-assigns spots.
    • Payment integration: Add a PaymentProcessor interface (OCP) with CreditCard, Cash, UPI implementations.
    • Floor preference: Strategy pattern for spot assignment (NearestToElevator, NearestToEntry).
