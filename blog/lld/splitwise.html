    Design: Splitwise

    March 2026 · 26 min read

    Splitwise is the go-to expense-sharing app, and a favorite in LLD interviews. It tests your understanding of Strategy pattern (multiple split modes), graph-based debt simplification, and clean class relationships. Let's design the core system.

    1Requirements

    Functional

    • Users can create groups (trip, apartment, etc.)
    • A user adds an expense specifying: payer, amount, participants, split type
    • Three split types: Equal, Exact (fixed amounts), Percentage
    • Track who owes whom and how much
    • Show balances per user (net amount owed/owing)
    • Simplify debts: minimize the number of transactions

    Non-Functional

    • Easy to add new split strategies (by weight, by shares, etc.)
    • Thread-safe balance updates

    2Identify Entities

    EntityResponsibility
    UserIdentity, contact info
    GroupCollection of users, holds expenses
    ExpenseRecords a payment: who paid, how much, how to split
    SplitStrategyCalculates each participant's share (interface)
    BalanceSheetTracks net balances between user pairs
    ExpenseManagerTop-level service, manages groups and expenses

    4Split Strategies OCP DIP

    class SplitStrategy {
public:
    virtual ~SplitStrategy() = default;
    // Returns map: user -> amount they owe
    virtual unordered_map<string, double> split(
        double totalAmount,
        const vector<string>& participants,
        const unordered_map<string, double>& params
    ) const = 0;
};

class EqualSplit : public SplitStrategy {
public:
    unordered_map<string, double> split(
        double totalAmount,
        const vector<string>& participants,
        const unordered_map<string, double>&
    ) const override {
        unordered_map<string, double> shares;
        double each = totalAmount / participants.size();
        for (const auto& user : participants)
            shares[user] = each;
        return shares;
    }
};

class ExactSplit : public SplitStrategy {
public:
    unordered_map<string, double> split(
        double totalAmount,
        const vector<string>& participants,
        const unordered_map<string, double>& exactAmounts
    ) const override {
        // Validate: amounts should sum to totalAmount
        double sum = 0;
        for (const auto& [user, amt] : exactAmounts) sum += amt;
        if (abs(sum - totalAmount) > 0.01)
            throw invalid_argument("Exact amounts don't sum to total");
        return exactAmounts;
    }
};

class PercentSplit : public SplitStrategy {
public:
    unordered_map<string, double> split(
        double totalAmount,
        const vector<string>& participants,
        const unordered_map<string, double>& percentages
    ) const override {
        double pctSum = 0;
        for (const auto& [user, pct] : percentages) pctSum += pct;
        if (abs(pctSum - 100.0) > 0.01)
            throw invalid_argument("Percentages don't sum to 100");

        unordered_map<string, double> shares;
        for (const auto& [user, pct] : percentages)
            shares[user] = totalAmount * pct / 100.0;
        return shares;
    }
};

    5Expense

    class Expense {
    static int nextId;
    int id;
    string description;
    string paidBy;
    double amount;
    vector<string> participants;
    unique_ptr<SplitStrategy> strategy;
    unordered_map<string, double> splitParams;

public:
    Expense(string desc, string payer, double amt,
            vector<string> parts,
            unique_ptr<SplitStrategy> strat,
            unordered_map<string, double> params = {})
        : id(nextId++), description(move(desc)), paidBy(move(payer)),
          amount(amt), participants(move(parts)),
          strategy(move(strat)), splitParams(move(params)) {}

    // Who owes how much
    unordered_map<string, double> getSplits() const {
        return strategy->split(amount, participants, splitParams);
    }

    const string& getPayer() const { return paidBy; }
    double getAmount() const { return amount; }
    const string& getDescription() const { return description; }
};
int Expense::nextId = 1;

    6Balance Sheet & Debt Simplification

    The balance sheet tracks directed debts: debts[A][B] = 50 means A owes B $50. When a new expense arrives, we update these balances.

    class BalanceSheet {
    // debts[debtor][creditor] = amount
    unordered_map<string, unordered_map<string, double>> debts;
    mutex mtx;

public:
    void recordExpense(const Expense& expense) {
        lock_guard<mutex> lock(mtx);
        auto splits = expense.getSplits();
        const string& payer = expense.getPayer();

        for (const auto& [user, amount] : splits) {
            if (user == payer) continue;  // payer doesn't owe themselves
            debts[user][payer] += amount;
            // Net off: if B already owed A, reduce that first
            if (debts[payer].count(user) && debts[payer][user] > 0) {
                double offset = min(debts[user][payer], debts[payer][user]);
                debts[user][payer] -= offset;
                debts[payer][user] -= offset;
            }
        }
    }

    // What does a specific user owe or is owed?
    void showBalances(const string& user) const {
        cout << "=== " << user << " ===" << endl;
        if (debts.count(user)) {
            for (const auto& [creditor, amt] : debts.at(user)) {
                if (amt > 0.01)
                    cout << "  Owes " << creditor << ": $" << amt << endl;
            }
        }
        for (const auto& [debtor, creditors] : debts) {
            if (creditors.count(user) && creditors.at(user) > 0.01)
                cout << "  " << debtor << " owes you: $" << creditors.at(user) << endl;
        }
    }

    // Minimize number of transactions using net-balance approach
    vector<tuple<string, string, double>> simplify() {
        lock_guard<mutex> lock(mtx);

        // Step 1: compute net balance per person
        unordered_map<string, double> net;
        for (const auto& [debtor, creditors] : debts) {
            for (const auto& [creditor, amt] : creditors) {
                net[debtor] -= amt;   // negative = owes money
                net[creditor] += amt; // positive = is owed money
            }
        }

        // Step 2: separate into creditors (+) and debtors (-)
        vector<pair<string, double>> creditors, debtors;
        for (const auto& [user, balance] : net) {
            if (balance > 0.01) creditors.push_back({user, balance});
            else if (balance < -0.01) debtors.push_back({user, -balance});
        }

        // Step 3: greedy matching
        vector<tuple<string, string, double>> transactions;
        int i = 0, j = 0;
        while (i < debtors.size() && j < creditors.size()) {
            double transfer = min(debtors[i].second, creditors[j].second);
            transactions.emplace_back(debtors[i].first, creditors[j].first, transfer);
            debtors[i].second -= transfer;
            creditors[j].second -= transfer;
            if (debtors[i].second < 0.01) i++;
            if (creditors[j].second < 0.01) j++;
        }

        return transactions;
    }
};

    Debt Simplification Example

    Before simplification (3 transactions):

    Alice $30→ Bob $20→ Charlie
    Alice $10→ Charlie

    After simplification (2 transactions, because Alice's net = −$40, Bob's net = $10, Charlie's net = +$30):

    Alice $10→ Bob
    Alice $30→ Charlie

    7ExpenseManager (Facade)

    class ExpenseManager {
    unordered_map<string, string> users;  // userId -> name
    unordered_map<string, vector<string>> groups;  // groupId -> userIds
    BalanceSheet balances;

    ExpenseManager() = default;
public:
    static ExpenseManager& getInstance() {
        static ExpenseManager instance;
        return instance;
    }

    void addUser(const string& id, const string& name) {
        users[id] = name;
    }

    void createGroup(const string& groupId, const vector<string>& members) {
        groups[groupId] = members;
    }

    void addExpense(const string& description, const string& payer,
                    double amount, const vector<string>& participants,
                    unique_ptr<SplitStrategy> strategy,
                    const unordered_map<string, double>& params = {}) {
        Expense expense(description, payer, amount, participants,
                        move(strategy), params);
        balances.recordExpense(expense);
    }

    void showBalances(const string& userId) {
        balances.showBalances(userId);
    }

    void simplifyAndShow() {
        auto txns = balances.simplify();
        cout << "=== Simplified Debts ===" << endl;
        for (const auto& [from, to, amt] : txns)
            cout << from << " pays " << to << ": $" << amt << endl;
    }
};

    8Putting It Together

    int main() {
    auto& mgr = ExpenseManager::getInstance();

    mgr.addUser("alice", "Alice");
    mgr.addUser("bob", "Bob");
    mgr.addUser("charlie", "Charlie");
    mgr.createGroup("trip", {"alice", "bob", "charlie"});

    // Alice pays $300 for hotel, split equally
    mgr.addExpense("Hotel", "alice", 300,
                   {"alice", "bob", "charlie"},
                   make_unique<EqualSplit>());

    // Bob pays $60 for dinner, exact split
    mgr.addExpense("Dinner", "bob", 60,
                   {"alice", "bob", "charlie"},
                   make_unique<ExactSplit>(),
                   {{"alice", 20}, {"bob", 25}, {"charlie", 15}});

    // Charlie pays $90 for fuel, percentage split
    mgr.addExpense("Fuel", "charlie", 90,
                   {"alice", "bob", "charlie"},
                   make_unique<PercentSplit>(),
                   {{"alice", 50}, {"bob", 30}, {"charlie", 20}});

    mgr.showBalances("alice");
    mgr.showBalances("bob");
    mgr.simplifyAndShow();
}

    Design Review

    Principle / PatternApplication
    SRPBalanceSheet manages debts. SplitStrategy computes shares. Expense is a data record. Each has one job.
    OCPAdd BySharesSplit, ByWeightSplit without touching Expense or BalanceSheet.
    DIPExpense depends on SplitStrategy interface, not concrete split classes.
    StrategySplit algorithms are interchangeable per expense.
    FacadeExpenseManager provides a simple API, hiding BalanceSheet and group mechanics.
    SingletonExpenseManager is the single entry point (appropriate for a service class).
