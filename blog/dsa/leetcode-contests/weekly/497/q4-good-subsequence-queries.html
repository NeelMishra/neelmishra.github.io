    Good Subsequence Queries

    Problem Statement

    You are given an integer array nums of length n and an integer p.

    A non-empty subsequence of nums is called good if:

    • Its length is strictly less than n.
    • The greatest common divisor (GCD) of its elements is exactly p.

    You are also given a 2D integer array queries of length q, where each queries[i] = [indi, vali] indicates that you should update nums[indi] to vali.

    After each query, determine whether there exists any good subsequence in the current array. Return the number of queries for which a good subsequence exists.

    Example 1:

    Input: nums = [4,8,12,16], p = 2, queries = [[0,3],[2,6]]
    Output: 1

    After the first query nums = [3,8,12,16] — no subsequence has GCD exactly 2. After the second, nums = [3,8,6,16] and [8, 6] works.

    Example 2:

    Input: nums = [4,5,7,8], p = 3, queries = [[0,6],[1,9],[2,3]]
    Output: 2

    Example 3:

    Input: nums = [5,7,9], p = 2, queries = [[1,4],[2,8]]
    Output: 0

    Constraints:

    • 2 <= n == nums.length <= 5 × 104
    • 1 <= nums[i] <= 5 × 104
    • 1 <= queries.length <= 5 × 104
    • queries[i] = [indi, vali]
    • 1 <= vali, p <= 5 × 104
    • 0 <= indi <= n - 1

    Intuition

    Only multiples of p can ever appear in a subsequence whose GCD is exactly p — any non-multiple drags the GCD to a non-multiple of p. So conceptually, after each update, ignore non-multiples and look at the GCD g of all the surviving multiples of p:

    • If g ≠ p, no subset of these elements can have GCD p (the GCD of any subset is a multiple of g, hence a strict multiple of p). Answer: no.
    • If g = p and at least one element is not a multiple of p, simply taking all multiples of p gives a subsequence of length < n with GCD p. Answer: yes.
    • If g = p and all n elements are multiples of p, we must drop at least one. We need some index i such that gcd(\text{left of } i, \text{right of } i) = p. By a small Erdős–Ko–Rado-style argument, when n > 6 this is always possible; only when n ≤ 6 do we have to actually try every drop.

    Updates are point assignments and the queried predicate (overall GCD over a range) composes well, so a segment tree over GCD — storing 0 in non-multiple slots and the value otherwise — is the natural data structure. Plus a counter cnt of how many positions are multiples of p.

    Brute Force

    Apply each query directly. After every update, scan the whole array and recompute the GCD of all multiples of p from scratch, then apply the same case analysis.

    class Solution {
public:
    int countGoodSubseq(vector<int>& nums, int p, vector<vector<int>>& queries) {
        int n = nums.size(), ans = 0;
        for (auto& q : queries) {
            nums[q[0]] = q[1];
            int g = 0, cnt = 0;
            for (int x : nums) if (x % p == 0) { g = __gcd(g, x); ++cnt; }
            if (g != p) continue;
            if (cnt < n) { ++ans; continue; }
            if (n > 6)   { ++ans; continue; }
            for (int i = 0; i < n; ++i) {
                int h = 0;
                for (int j = 0; j < n; ++j) if (j != i) h = __gcd(h, nums[j]);
                if (h == p) { ++ans; break; }
            }
        }
        return ans;
    }
};

    Complexity: Time O(q · n · log V) where V = max(nums) — at the constraint extremes ≈ 5×10⁴ × 5×10⁴ × 17 ≈ 4×10¹⁰. Too slow.

    Optimal Approach

    Maintain a segment tree of range GCDs over a transformed array a[i]: a[i] = nums[i] when p | nums[i], else 0. We use the convention gcd(0, x) = x so zeros are neutral. Maintain a counter cnt for the number of non-zero positions (i.e. multiples of p).

    Per query (idx, val):

    1. Decrement cnt if the old nums[idx] was a multiple of p; point-update the segment tree to 0.
    2. If val is a multiple of p, point-update to val and increment cnt; otherwise leave it at 0.
    3. Read the global GCD from the root. If it equals p: the answer is yes when cnt < n (drop a non-multiple) or when n > 6; otherwise (small n, all multiples) brute-force which single index to drop using two range queries ([1..i−1] and [i+1..n]) and check whether their merged GCD equals p.
    class Solution {
    vector<int> tr;
    int N;
    void upd(int u, int l, int r, int x, int v) {
        if (l == r) { tr[u] = v; return; }
        int m = (l + r) >> 1;
        if (x <= m) upd(u<<1, l, m, x, v);
        else        upd(u<<1|1, m+1, r, x, v);
        tr[u] = __gcd(tr[u<<1], tr[u<<1|1]);
    }
    int qry(int u, int l, int r, int ql, int qr) {
        if (ql > qr || qr < l || ql > r) return 0;
        if (ql <= l && r <= qr) return tr[u];
        int m = (l + r) >> 1;
        return __gcd(qry(u<<1, l, m, ql, qr), qry(u<<1|1, m+1, r, ql, qr));
    }
public:
    int countGoodSubseq(vector<int>& nums, int p, vector<vector<int>>& queries) {
        int n = nums.size();
        N = n; tr.assign(4*n, 0);
        int cnt = 0;
        for (int i = 0; i < n; ++i) if (nums[i] % p == 0) {
            upd(1, 1, n, i+1, nums[i]); ++cnt;
        }
        int ans = 0;
        for (auto& q : queries) {
            int idx = q[0], val = q[1];
            if (nums[idx] % p == 0) { upd(1, 1, n, idx+1, 0); --cnt; }
            if (val % p == 0)      { upd(1, 1, n, idx+1, val); ++cnt; }
            nums[idx] = val;
            if (tr[1] != p) continue;
            if (cnt < n || n > 6) { ++ans; continue; }
            for (int i = 1; i <= n; ++i) {
                int L = qry(1, 1, n, 1, i-1);
                int R = qry(1, 1, n, i+1, n);
                if (__gcd(L, R) == p) { ++ans; break; }
            }
        }
        return ans;
    }
};

    Why it works: The two facts behind the case split are: (i) replacing every non-multiple of p by 0 doesn’t change the GCD over multiples of p (with gcd(0, x) = x); and (ii) when n > 6 and every element is a multiple of p with overall GCD p, you can always remove one element while keeping the GCD p. (Sketch: if every removal raised the GCD, each element would be the unique “blocker” for some prime that divides every other element — this forces n ≤ 6.) For the small remainder case we exhaustively try each removal in O(n · log n) per query.

    Complexity: Time O((n + q) · log n · log V) where log V comes from each gcd. Space O(n).

    Dry Run

    Take Example 1: nums = [4,8,12,16], p = 2, queries [[0,3],[2,6]]. Initially every element is a multiple of 2, so the segment tree stores [4, 8, 12, 16], the root GCD is 4, and cnt = 4.

    1. Query [0, 3]: Old nums[0] = 4 was a multiple of 2, so update tree[0] to 0, cnt = 3. New value 3 is not a multiple of 2, leave 0. Tree leaves: [0, 8, 12, 16], root GCD = gcd(8,12,16) = 4 ≠ 2. No good subsequence.
    2. Query [2, 6]: Old nums[2] = 12 was a multiple of 2 — set tree[2] to 0, cnt = 2. New val = 6 is a multiple of 2; set tree[2] to 6, cnt = 3. Tree leaves: [0, 8, 6, 16], root GCD = gcd(8, 6, 16) = 2 = p ✓. Since cnt = 3 < n = 4, we can pick all multiples of 2 (e.g. [8, 6, 16] or even just [8, 6]) and the GCD is exactly 2. Good subsequence exists. Increment ans.

    Final answer: 1, matching the expected output.

    Complexity Recap

    ApproachTimeSpaceNotes
    Brute (rescan per query)O(q · n · log V)O(1)4×10¹⁰ ops at constraint maxima — TLE.
    Optimal (GCD segment tree)O((n + q) · log n · log V)O(n)Small-n case (n ≤ 6) adds only constant extra factor.

    Edge Cases & Pitfalls

    • GCD identity: Use gcd(0, x) = x so that “empty” segment-tree leaves (non-multiples of p) act as the identity. Initialising leaves to 0 is correct only with this convention.
    • Strictly less than n: The full array itself does not count, so when every element is a multiple of p you must verify some single element can be dropped while keeping GCD = p.
    • Threshold n > 6: The constant 6 is tight; replacing it with a smaller bound breaks correctness (you may answer “no” on small arrays where a witness exists), and replacing it with a larger bound is just slower.
    • GCD ≠ p short-circuit: If the global GCD over multiples of p is kp with k ≥ 2, no subset can lower it to p; skip the small-n loop entirely.
    • Repeated index updates: Always undo the old contribution to cnt and the segment tree before applying the new value, even if val == nums[idx].
    • Non-multiple to non-multiple update: Both branches are no-ops on the tree but you still must overwrite nums[idx] for future queries.
    • p = 1: Every integer is a multiple of 1; the “non-multiple” case never triggers, so the small-n branch always runs when n ≤ 6.

    class Solution {
    vector<int> tr;
    void upd(int u, int l, int r, int x, int v) {
        if (l == r) { tr[u] = v; return; }
        int m = (l + r) >> 1;
        if (x <= m) upd(u<<1, l, m, x, v);
        else        upd(u<<1|1, m+1, r, x, v);
        tr[u] = __gcd(tr[u<<1], tr[u<<1|1]);
    }
    int qry(int u, int l, int r, int ql, int qr) {
        if (ql > qr || qr < l || ql > r) return 0;
        if (ql <= l && r <= qr) return tr[u];
        int m = (l + r) >> 1;
        return __gcd(qry(u<<1, l, m, ql, qr), qry(u<<1|1, m+1, r, ql, qr));
    }
public:
    int countGoodSubseq(vector<int>& nums, int p, vector<vector<int>>& queries) {
        int n = nums.size();
        tr.assign(4 * n, 0);
        int cnt = 0;
        for (int i = 0; i < n; ++i) {
            if (nums[i] % p == 0) {
                upd(1, 1, n, i + 1, nums[i]);
                ++cnt;
            }
        }
        int ans = 0;
        for (auto& q : queries) {
            int idx = q[0], val = q[1];
            if (nums[idx] % p == 0) { upd(1, 1, n, idx + 1, 0); --cnt; }
            if (val % p == 0)      { upd(1, 1, n, idx + 1, val); ++cnt; }
            nums[idx] = val;
            if (tr[1] != p) continue;
            if (cnt < n || n > 6) { ++ans; continue; }
            for (int i = 1; i <= n; ++i) {
                int L = qry(1, 1, n, 1, i - 1);
                int R = qry(1, 1, n, i + 1, n);
                if (__gcd(L, R) == p) { ++ans; break; }
            }
        }
        return ans;
    }
};
