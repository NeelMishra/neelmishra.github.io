    Ordinary Least Squares in C++

    July 2026 · 18 min read

    Ordinary least squares minimizes squared error. This implementation forms the normal equations and solves them with Gaussian elimination. It is compact and educational, though QR or SVD is preferable for production numerical stability.

    Linear Solver

    using Vector = vector<double>;
using Matrix = vector<Vector>;

Vector solve_linear(Matrix A, Vector b) {
    int n = (int)b.size();
    for (int col = 0; col < n; ++col) {
        int piv = col;
        for (int r = col + 1; r < n; ++r)
            if (fabs(A[r][col]) > fabs(A[piv][col])) piv = r;
        swap(A[col], A[piv]);
        swap(b[col], b[piv]);
        double div = A[col][col];
        for (int c = col; c < n; ++c) A[col][c] /= div;
        b[col] /= div;
        for (int r = 0; r < n; ++r) if (r != col) {
            double m = A[r][col];
            for (int c = col; c < n; ++c) A[r][c] -= m * A[col][c];
            b[r] -= m * b[col];
        }
    }
    return b;
}

    Fit OLS

    Vector fit_ols(const Matrix& X, const Vector& y) {
    int n = (int)X.size();
    int d = (int)X[0].size() + 1; // intercept
    Matrix xtx(d, Vector(d, 0.0));
    Vector xty(d, 0.0);

    for (int i = 0; i < n; ++i) {
        Vector row(d, 1.0);
        for (int j = 1; j < d; ++j) row[j] = X[i][j - 1];
        for (int a = 0; a < d; ++a) {
            xty[a] += row[a] * y[i];
            for (int b = 0; b < d; ++b) xtx[a][b] += row[a] * row[b];
        }
    }
    return solve_linear(xtx, xty);
}

    Prediction

    double predict(const Vector& w, const Vector& x) {
    double yhat = w[0];
    for (size_t j = 0; j < x.size(); ++j) yhat += w[j + 1] * x[j];
    return yhat;
}
